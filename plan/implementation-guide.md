## 1. 核心注解实现

### 1.1 FeatureProvider 注解

**文件**: `shared/src/commonMain/kotlin/com/wgt/architecture/di/annotations/FeatureProvider.kt`

```kotlin
package com.wgt.architecture.di.annotations

import com.wgt.architecture.di.Lifecycle
import kotlin.annotation.AnnotationRetention
import kotlin.annotation.AnnotationTarget

/**
 * 标记一个类为Feature提供者
 * KSP将自动生成注册代码和扩展属性
 */
@Target(AnnotationTarget.CLASS)
@Retention(AnnotationRetention.SOURCE)
annotation class FeatureProvider(
    /**
     * Feature的属性名称，默认使用类名转换
     * 例如: PermissionFeature -> permission
     */
    val propertyName: String = "",
    
    /**
     * 生命周期类型
     */
    val lifecycle: Lifecycle = Lifecycle.SINGLETON
)
```

### 2.3 主处理器

**文件**: `ksp-processor/src/main/kotlin/com/wgt/ksp/FeatureProcessor.kt`

```kotlin
package com.wgt.ksp

import com.google.devtools.ksp.processing.*
import com.google.devtools.ksp.symbol.*
import com.squareup.kotlinpoet.*
import com.squareup.kotlinpoet.ksp.writeTo

class FeatureProcessor(
    private val codeGenerator: CodeGenerator,
    private val logger: KSPLogger
) : SymbolProcessor {
    
    private val features = mutableListOf<FeatureInfo>()
    
    override fun process(resolver: Resolver): List<KSAnnotated> {
        val symbols = resolver.getSymbolsWithAnnotation(
            "com.wgt.architecture.di.annotations.FeatureProvider"
        )
        
        symbols.filterIsInstance<KSClassDeclaration>().forEach { classDecl ->
            processFeatureClass(classDecl)
        }
        
        return emptyList()
    }
    
    override fun finish() {
        if (features.isEmpty()) return
        
        generateExtensionProperties()
        generateRegistrationFunction()
    }
    
    private fun processFeatureClass(classDecl: KSClassDeclaration) {
        val annotation = classDecl.annotations.first {
            it.shortName.asString() == "FeatureProvider"
        }
        
        val propertyName = annotation.arguments
            .find { it.name?.asString() == "propertyName" }
            ?.value as? String
            ?: generatePropertyName(classDecl.simpleName.asString())
        
        val lifecycle = annotation.arguments
            .find { it.name?.asString() == "lifecycle" }
            ?.value?.toString() ?: "SINGLETON"
        
        features.add(
            FeatureInfo(
                className = classDecl.qualifiedName?.asString() ?: return,
                simpleName = classDecl.simpleName.asString(),
                propertyName = propertyName,
                lifecycle = lifecycle,
                packageName = classDecl.packageName.asString()
            )
        )
        
        logger.info("Processed feature: ${classDecl.simpleName.asString()} -> $propertyName")
    }
    
    private fun generatePropertyName(className: String): String {
        return className
            .removeSuffix("Feature")
            .replaceFirstChar { it.lowercase() }
    }
    
    private fun generateExtensionProperties() {
        val fileSpec = FileSpec.builder(
            packageName = "com.wgt.architecture.di.generated",
            fileName = "FeatureExtensions"
        ).apply {
            addFileComment("Auto-generated by KSP. Do not modify.")
            
            features.forEach { feature ->
                addProperty(
                    PropertySpec.builder(
                        name = feature.propertyName,
                        type = ClassName.bestGuess(feature.className)
                    )
                    .receiver(ClassName("com.wgt.architecture.manager.claim", "IFeatureManager"))
                    .delegate("inject()")
                    .addKdoc("Auto-generated extension property for ${feature.simpleName}")
                    .build()
                )
            }
            
            addImport("com.wgt.architecture.di", "inject")
        }.build()
        
        fileSpec.writeTo(codeGenerator, Dependencies(false))
    }
    
    private fun generateRegistrationFunction() {
        val fileSpec = FileSpec.builder(
            packageName = "com.wgt.architecture.di.generated",
            fileName = "FeatureRegistry"
        ).apply {
            addFileComment("Auto-generated by KSP. Do not modify.")
            
            val functionSpec = FunSpec.builder("registerAllFeatures")
                .addKdoc("Auto-generated feature registration function")
                .apply {
                    features.forEach { feature ->
                        addStatement(
                            "registerFeature<%T>(Lifecycle.%L) { %T() }",
                            ClassName.bestGuess(feature.className),
                            feature.lifecycle,
                            ClassName.bestGuess(feature.className)
                        )
                    }
                }
                .build()
            
            addFunction(functionSpec)
            
            addImport("com.wgt.architecture.di", "Lifecycle")
            addImport("com.wgt.architecture.feature", "registerFeature")
        }.build()
        
        fileSpec.writeTo(codeGenerator, Dependencies(false))
    }
}

data class FeatureInfo(
    val className: String,
    val simpleName: String,
    val propertyName: String,
    val lifecycle: String,
    val packageName: String
)
```

## 5. 应用初始化

```kotlin
package com.wgt

import android.app.Application
import com.wgt.architecture.di.generated.registerAllFeatures
import com.wgt.architecture.manager.InitManager
import com.wgt.architecture.manager.claim.feature
import com.wgt.architecture.manager.manager
import kotlinx.coroutines.GlobalScope
import kotlinx.coroutines.launch

class MediaApplication : Application() {
    override fun onCreate() {
        super.onCreate()
        
        // 1. 初始化Manager系统
        InitManager()
        
        // 2. 自动注册所有Feature（KSP生成）
        registerAllFeatures()
        
        // 3. 激活FeatureManager
        GlobalScope.launch {
            manager.feature.initialize()
            manager.feature.activate()
        }
    }
}
```

## 6. 使用示例

### 6.1 基本使用

```kotlin
import com.wgt.architecture.manager.claim.feature
import com.wgt.architecture.manager.manager

class MyViewModel {
    // 通过生成的扩展属性访问Feature
    private val permissionFeature = manager.feature.permission
    
    suspend fun requestCameraPermission() {
        val result = permissionFeature.requestPermission(
            PermissionType.CAMERA
        )
        
        when (result) {
            is PermissionResult.Granted -> {
                // 权限已授予
            }
            is PermissionResult.Denied -> {
                // 权限被拒绝
            }
        }
    }
}
```

## 8. 故障排查

### 8.1 常见问题

**问题1**: 生成的代码找不到
- 检查 KSP 插件是否正确配置
- 确认 `kotlin.srcDir` 配置正确
- 运行 `./gradlew clean build`

**问题2**: 编译错误 "Unresolved reference"
- 确保先运行 KSP 任务: `./gradlew kspCommonMainKotlinMetadata`
- 检查依赖顺序配置

**问题3**: Feature 未注册
- 确认类上有 `@FeatureProvider` 注解
- 检查 KSP 日志输出
- 验证 `registerAllFeatures()` 被调用

### 8.2 调试技巧

```kotlin
// 在应用启动时打印注册信息
fun debugFeatureRegistration() {
    val registeredTypes = DependencyContainer.getRegisteredTypes()
    registeredTypes.forEach { type ->
        println("Registered: ${type.simpleName}")
    }
}
```
